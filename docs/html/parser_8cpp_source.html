36Token *Parser::parse_expect(Lexer &lexer, TokenType expected) {
37 Token *tok = lexer.next();
38 if (tok->type() != expected) {
39 ERR_WARGS(ErrType::Syntax,
40 "parse_expect: expected %d, got %s `%s`",
41 (int)expected, tok->to_str().c_str(), tok->lexeme().c_str());
42 }
43 return tok;

46Token *Parser::parse_expect_keyword(Lexer &lexer, std::string expected) {
47 Token *tok = lexer.next();
48
49 if (tok->type() != TokenType::Keyword) {
50 ERR_WARGS(ErrType::Syntax,
51 "parse_expect_keyword: %s `%s` is not a keyword",
52 tok->to_str().c_str(), tok->lexeme().c_str());
53 }
54 if (tok->lexeme() != expected) {
55 ERR_WARGS(ErrType::Syntax,
56 "parse_expect_keyword: expected keyword `%s`, got %s `%s`",
57 expected.c_str(), tok->to_str().c_str(), tok->lexeme().c_str());
58 }
59
60 return tok;

73static std::vector<std::unique_ptr<Expr>> parse_comma_sep_exprs(Lexer &lexer) {
74 std::vector<std::unique_ptr<Expr>> exprs;
75
76 (void)Parser::parse_expect(lexer, TokenType::Lparen);
77
78 while (1) {
79 // Only needed if no arguments are provided.
80 if (lexer.peek()->type() == TokenType::Rparen) {
81 break;
82 }
83 exprs.push_back(std::unique_ptr<Expr>(Parser::parse_expr(lexer)));
84 if (lexer.peek()->type() == TokenType::Comma) {
85 (void)Parser::parse_expect(lexer, TokenType::Comma);
86 }
87 else {
88 break;
89 }
90 }
91
92 Parser::parse_expect(lexer, TokenType::Rparen);
93 return exprs;
94}

96static std::optional<std::vector<std::unique_ptr<Expr>>> try_parse_funccall(Lexer &lexer) {
97 if (lexer.peek()->type() == TokenType::Lparen) {

105static Expr *parse_primary_expr(Lexer &lexer) {
106 Token *tok = lexer.next();
107
108 switch (tok->type()) {
109 case TokenType::Ident: {
110 auto exprs = try_parse_funccall(lexer);
111
112 // We are parsing a function call.
113 if (exprs.has_value()) {
114 return new ExprFuncCall(std::make_unique<Token>(*tok), std::move(exprs.value()));
115 }
116
117 return new ExprIdent(std::make_unique<Token>(*tok));
118 } break;
119 case TokenType::Intlit: {
120 return new ExprIntLit(std::make_unique<Token>(*tok));
121 } break;
122 case TokenType::Strlit: {
123 return new ExprStrLit(std::make_unique<Token>(*tok));
124 } break;
125 default:
126 assert(false && "parse_primary_expr: invalid primary expression");
127 }
128}

130static Expr *parse_multiplicative_expr(Lexer &lexer) {
131 Expr *lhs = parse_primary_expr(lexer);
132 Token *cur = lexer.peek();
133 while (cur && (cur->type() == TokenType::Asterisk
134 || cur->type() == TokenType::Forwardslash)) {
135 Token *op = lexer.next();
136 Expr *rhs = parse_primary_expr(lexer);
137 lhs = new ExprBinary(std::unique_ptr<Expr>(lhs),
138 std::make_unique<Token>(*op),
139 std::unique_ptr<Expr>(rhs));
140 cur = lexer.peek();
141 }
142 return lhs;
143}

145static Expr *parse_additive_expr(Lexer &lexer) {
146 Expr *lhs = parse_multiplicative_expr(lexer);

160static Expr *parse_equalitative_expr(Lexer &lexer) {
161 Expr *lhs = parse_additive_expr(lexer);
162 Token *cur = lexer.peek();

164 || cur->type() == TokenType::Greaterthan_Equals
165 || cur->type() == TokenType::Greaterthan

206std::unique_ptr<StmtIf> Parser::parse_stmt_if(Lexer &lexer) {
207 (void)Parser::parse_expect_keyword(lexer, COMMON_EARLKW_IF);
208
209 Expr *expr = Parser::parse_expr(lexer);
210 std::unique_ptr<StmtBlock> block = Parser::parse_stmt_block(lexer);
211
212 // Handle the `else if` or `else` blocks if applicable
213 std::optional<std::unique_ptr<StmtBlock>> else_ = {};
214 Token *tok1 = lexer.peek();
215 Token *tok2 = lexer.peek(1);
216
217 bool tok1_else = tok1->type() == TokenType::Keyword && tok1->lexeme() == COMMON_EARLKW_ELSE;
218 bool tok2_if = tok2->type() == TokenType::Keyword && tok2->lexeme() == COMMON_EARLKW_IF;
219
220 if (tok1_else && tok2_if) {
221 lexer.discard();
222 std::vector<std::unique_ptr<Stmt>> tmp;
223 std::unique_ptr<StmtIf> nested_if = parse_stmt_if(lexer);
224 tmp.push_back(std::move(nested_if));
225 else_ = std::make_unique<StmtBlock>(std::move(tmp));
226 }
227 else if (tok1_else) {
228 Parser::parse_expect_keyword(lexer, COMMON_EARLKW_ELSE);
229 else_ = parse_stmt_block(lexer);
230 }
231
232 return std::make_unique<StmtIf>(std::unique_ptr<Expr>(expr),
233 std::move(block),
234 std::move(else_));
235}

257std::unique_ptr<StmtBlock> Parser::parse_stmt_block(Lexer &lexer) {
258 (void)Parser::parse_expect(lexer, TokenType::Lbrace);
259
260 std::vector<std::unique_ptr<Stmt>> stmts;
261
262 while (lexer.peek()->type() != TokenType::Rbrace) {
263 std::unique_ptr<Stmt> stmt = Parser::parse_stmt(lexer);
264 stmts.push_back(std::move(stmt));

269 return std::make_unique<StmtBlock>(std::move(stmts));

272std::vector<std::pair<std::unique_ptr<Token>, std::unique_ptr<Token>>>
273parse_stmt_def_args(Lexer &lexer) {
274 std::vector<std::pair<std::unique_ptr<Token>, std::unique_ptr<Token>>> args;
275
276 (void)Parser::parse_expect(lexer, TokenType::Lparen);
277 while (lexer.peek()->type() != TokenType::Rparen) {
278 Token *id = Parser::parse_expect(lexer, TokenType::Ident);
279 (void)Parser::parse_expect(lexer, TokenType::Colon);
280 Token *type = Parser::parse_expect_type(lexer);
281 auto pair = std::make_pair(std::make_unique<Token>(*id), std::make_unique<Token>(*type));
282 args.push_back(std::move(pair));
283
284 if (lexer.peek()->type() == TokenType::Comma)
285 lexer.discard();
286 }
287 (void)Parser::parse_expect(lexer, TokenType::Rparen);
288
289 return args;
290}

344std::unique_ptr<Stmt> Parser::parse_stmt(Lexer &lexer) {
345 Token *tok = lexer.peek();
346
347 switch (tok->type()) {
348 case TokenType::Keyword: {
349 if (tok->lexeme() == COMMON_EARLKW_LET) {
350 return parse_stmt_let(lexer);
351 }
352 else if (tok->lexeme() == COMMON_EARLKW_DEF) {
353 return parse_stmt_def(lexer);
354 }
355 else if (tok->lexeme() == COMMON_EARLKW_IF) {
356 return parse_stmt_if(lexer);
357 }
358 else if (tok->lexeme() == COMMON_EARLKW_RETURN) {
359 return parse_stmt_return(lexer);
360 }
361 else if (tok->lexeme() == COMMON_EARLKW_WHILE) {
362 return parse_stmt_while(lexer);
363 }
364 else if (tok->lexeme() == COMMON_EARLKW_FOR) {
365 return parse_stmt_for(lexer);
366 }
367 else {
368 assert(false && "parse_stmt: invalid keyword");
369 }
370 } break;
371 case TokenType::Ident: {
372 if (lexer.peek(1)->type() == TokenType::Lparen) {
373 return parse_stmt_expr(lexer);
374 }
375 return parse_stmt_mut(lexer);
376 } break;
377 default:
378 assert(false && "parse_stmt: invalid statement");
379 }
380}
