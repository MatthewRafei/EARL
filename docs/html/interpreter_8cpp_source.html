47EarlTy::Type Interpreter::ExprEvalResult::get_earl_type(Ctx &ctx) {
48 if (m_expr_term_type == ExprTermType::Ident) {
49 Token *tok = std::any_cast<Token *>(m_expr_value);
50 auto *var = ctx.get_registered_earlvar(tok->lexeme());
51 return var->m_type;
52 }
53
54 switch (m_expr_term_type) {
55 case ExprTermType::Int_Literal: return EarlTy::Type::Int;
56 case ExprTermType::Str_Literal: return EarlTy::Type::Str;
57 default:
58 ERR_WARGS(ErrType::Fatal, "ExprTermType `%d` is not a valid EARL type",
59 static_cast<int>(m_expr_term_type));
60 }
61}

63static Interpreter::ExprEvalResult
64eval_user_defined_function(ExprFuncCall *expr,
65 std::vector<Interpreter::ExprEvalResult>
66 user_params,
67 Ctx &ctx) {
68
69 auto *func = ctx.get_registered_earlfunc(expr->m_id->lexeme());
70
71 ctx.set_current_earlfunc(func);
72
73 for (size_t i = 0; i < expr->m_params.size(); ++i) {
74 Interpreter::ExprEvalResult user_param = user_params[i];
75 func->m_args[i]->set_value(user_param.value());
76 ctx.register_earlvar(func->m_args[i]);
77 }
78
79 Interpreter::ExprEvalResult blockresult = eval_stmt_block(func->m_block, ctx);
80
81 ctx.unset_current_earlfunc();
82
83 return blockresult;
84}

86Interpreter::ExprEvalResult eval_expr_funccall(ExprFuncCall *expr, Ctx &ctx) {
87 std::vector<Interpreter::ExprEvalResult> arg_evals;
88 for (size_t i = 0; i < expr->m_params.size(); ++i) {
89 std::unique_ptr<Expr> &e = expr->m_params[i];
90 Interpreter::ExprEvalResult param = Interpreter::eval_expr(e.get(), ctx);
91 arg_evals.push_back(param);
92 }
93
94 if (Intrinsics::is_intrinsic_function(expr->m_id->lexeme())) {
95 return Intrinsics::run_intrinsic_function(expr, arg_evals, ctx);
96 }
97 return eval_user_defined_function(expr, arg_evals, ctx);
98}

100Interpreter::ExprEvalResult eval_expr_term(ExprTerm *expr, Ctx &ctx) {
101 switch (expr->get_term_type()) {
102 case ExprTermType::Ident: {
103 ExprIdent *ident = dynamic_cast<ExprIdent *>(expr);
104 EarlVar *stored = ctx.get_registered_earlvar(ident->m_tok->lexeme());
105 return Interpreter::ExprEvalResult {
106 stored->m_value,
107 ident->get_term_type(),
108 stored->m_type,
109 Interpreter::LiteralResult::Result { stored },
110 };
111 } break;
112 case ExprTermType::Int_Literal: {
113 ExprIntLit *intlit = dynamic_cast<ExprIntLit *>(expr);
114 return Interpreter::ExprEvalResult {
115 std::stoi(intlit->m_tok->lexeme()),
116 intlit->get_term_type(),
117 EarlTy::Type::Int,
118 Interpreter::LiteralResult::Result { intlit->m_tok.get() },
119 };
120 } break;
121 case ExprTermType::Str_Literal: {
122 ExprStrLit *strlit = dynamic_cast<ExprStrLit *>(expr);
123 return Interpreter::ExprEvalResult {
124 strlit->m_tok->lexeme(),
125 strlit->get_term_type(),
126 EarlTy::Type::Str,
127 Interpreter::LiteralResult::Result { strlit->m_tok.get() },
128 };
129 } break;
130 case ExprTermType::Func_Call: {
131 return eval_expr_funccall(dynamic_cast<ExprFuncCall *>(expr), ctx);
132 } break;
133 default:
134 ERR_WARGS(ErrType::Fatal, "%d is not a valid expression term type is not valid",
135 static_cast<int>(expr->get_term_type()));
136 }
137 return Interpreter::ExprEvalResult{};
138}

140Interpreter::ExprEvalResult eval_expr_bin(ExprBinary *expr, Ctx &ctx) {
141 Interpreter::ExprEvalResult lhs = Interpreter::eval_expr(expr->m_lhs.get(), ctx);
142 Interpreter::ExprEvalResult rhs = Interpreter::eval_expr(expr->m_rhs.get(), ctx);
143
144 if (!EarlTy::earlvar_type_compat(lhs.m_earl_type, rhs.m_earl_type)) {
145 ERR_WARGS(ErrType::ERR_FATAL, "type (%d) is not compatable with type (%d)",
146 static_cast<int>(lhs.m_earl_type), static_cast<int>(rhs.m_earl_type));
147 }
148
149 switch (expr->m_op->type()) {
150 case TokenType::Plus: {
151 return Interpreter::ExprEvalResult {
152 std::any_cast<int>(lhs.value()) + std::any_cast<int>(rhs.value()),
153 ExprTermType::Int_Literal,
154 lhs.m_earl_type,
155 Interpreter::LiteralResult::Result {false},
156 };
157 } break;
158 case TokenType::Minus: {
159 return Interpreter::ExprEvalResult {
160 std::any_cast<int>(lhs.value()) - std::any_cast<int>(rhs.value()),
161 ExprTermType::Int_Literal,
162 lhs.m_earl_type,
163 Interpreter::LiteralResult::Result {false},
164 };
165 } break;
166 case TokenType::Asterisk: {
167 return Interpreter::ExprEvalResult {
168 std::any_cast<int>(lhs.value()) * std::any_cast<int>(rhs.value()),
169 ExprTermType::Int_Literal,
170 lhs.m_earl_type,
171 Interpreter::LiteralResult::Result {false},
172 };
173 } break;
174 case TokenType::Forwardslash: {
175 return Interpreter::ExprEvalResult {
176 std::any_cast<int>(lhs.value()) / std::any_cast<int>(rhs.value()),
177 ExprTermType::Int_Literal,
178 lhs.m_earl_type,
179 Interpreter::LiteralResult::Result {false},
180 };
181 } break;
182 case TokenType::Double_Equals: {
183 return Interpreter::ExprEvalResult {
184 std::any_cast<int>(lhs.value()) == std::any_cast<int>(rhs.value()),
185 ExprTermType::Int_Literal,
186 lhs.m_earl_type,
187 Interpreter::LiteralResult::Result {false},
188 };
189 } break;
190 case TokenType::Bang_Equals: {
191 return Interpreter::ExprEvalResult {
192 std::any_cast<int>(lhs.value()) != std::any_cast<int>(rhs.value()),
193 ExprTermType::Int_Literal,
194 lhs.m_earl_type,
195 Interpreter::LiteralResult::Result {false},
196 };
197 } break;
198 default:
199 ERR_WARGS(ErrType::Fatal, "%s is not a valid binary operator", expr->m_op->lexeme().c_str());
200 }
201
202 // Unreachable
203 return Interpreter::ExprEvalResult {};
204}

206Interpreter::ExprEvalResult Interpreter::eval_expr(Expr *expr, Ctx &ctx) {
207 switch (expr->get_type()) {
208 case ExprType::Term: {
209 return eval_expr_term(dynamic_cast<ExprTerm *>(expr), ctx);
210 } break;
211 case ExprType::Binary: {
212 return eval_expr_bin(dynamic_cast<ExprBinary *>(expr), ctx);
213 } break;
214 default:
215 ERR_WARGS(ErrType::Fatal, "expression type %d is not a valid expression",
216 static_cast<int>(expr->get_type()));
217 }
218 return Interpreter::ExprEvalResult{};
219}

221Interpreter::ExprEvalResult eval_stmt_let(StmtLet *stmt, Ctx &ctx) {
222 const std::string &id = stmt->m_id->lexeme();
223
224 if (ctx.is_registered_earlvar(id)) {
225 ERR_WARGS(ErrType::Redeclared, "variable `%s` is already defined", id.c_str());
226 }
227
228 // The `let` type binding i.e., let x: <TYPE> = ...;
229 EarlTy::Type binding_type = EarlTy::of_str(stmt->m_type->lexeme());
230
231 Interpreter::ExprEvalResult expr_eval = Interpreter::eval_expr(stmt->m_expr.get(), ctx);
232
233 // The type of the right side of the equals sign
234 EarlTy::Type rval_type = expr_eval.m_earl_type;
235
236 if (!EarlTy::earlvar_type_compat(binding_type, rval_type)) {
237 ERR_WARGS(ErrType::ERR_FATAL, "type (%d) is not compatable with type (%d)",
238 static_cast<int>(binding_type), static_cast<int>(rval_type));
239 }
240
241 EarlVar *var = new EarlVar(stmt->m_id.get(), binding_type, false, expr_eval.m_expr_value);
242 ctx.register_earlvar(var);
243
244 return Interpreter::ExprEvalResult{};
245}

251Interpreter::ExprEvalResult eval_stmt_block(StmtBlock *block, Ctx &ctx) {
252 Interpreter::ExprEvalResult result{};
253
254 ctx.push_scope();
255 for (auto &stmt : block->m_stmts) {
256 result = eval_stmt(stmt.get(), ctx);
257 if (result.value().has_value())
258 break;
259 }
260 ctx.pop_scope();
261
262 return result;
263}

265// When we hit a statement `def` (a function declaration),
266// we do not actually want to execute this function.
267// We just want to add it to the global context so it
268// can be called later from either a statement expression
269// or a right-hand-side assignment.
270Interpreter::ExprEvalResult eval_stmt_def(StmtDef *stmt, Ctx &ctx) {
271 std::vector<EarlVar *> args;
272 for (auto &arg : stmt->m_args) {
273 Token *id = arg.first.get();
274 EarlTy::Type type = EarlTy::of_str(arg.second->lexeme());
275 args.push_back(new EarlVar(id, type, false, nullptr));
276 }
277
278 auto *func = new EarlFunc::Func(stmt->m_id.get(),
279 EarlTy::of_str(stmt->m_rettype->lexeme()),
280 std::move(args),
281 stmt->m_block.get());
282
283 ctx.register_earlfunc(func);
284
285 return Interpreter::ExprEvalResult{};
286}

288Interpreter::ExprEvalResult eval_stmt_if(StmtIf *stmt, Ctx &ctx) {
289 Interpreter::ExprEvalResult expr_result = Interpreter::eval_expr(stmt->m_expr.get(), ctx);
290 Interpreter::ExprEvalResult result{};
291
292 if (std::any_cast<bool>(expr_result.value()) == true) {
293 result = eval_stmt_block(stmt->m_block.get(), ctx);
294 }
295 else if (stmt->m_else.has_value()) {
296 result = eval_stmt_block(stmt->m_else.value().get(), ctx);
297 }
298
299 return result;
300}

306Interpreter::ExprEvalResult eval_stmt_mut(StmtMut *stmt, Ctx &ctx) {
307 Interpreter::ExprEvalResult left = Interpreter::eval_expr(stmt->m_left.get(), ctx);
308 Interpreter::ExprEvalResult right = Interpreter::eval_expr(stmt->m_right.get(), ctx);
309
310 EarlVar *var = std::get<EarlVar *>(left.m_literal_result.m_value);
311 assert(ctx.is_registered_earlvar(var->m_id->lexeme()));
312
313 if (!EarlTy::earlvar_type_compat(left.m_earl_type, right.m_earl_type)) {
314 ERR_WARGS(ErrorType::Fatal,
315 "type %d is not compatable with type %d",
316 static_cast<int>(left.m_earl_type),
317 static_cast<int>(right.m_earl_type));
318 }
319
320 var = ctx.get_registered_earlvar(var->m_id->lexeme());
321 var->set_value(right.value());
322
323 return Interpreter::ExprEvalResult{};
324}

326Interpreter::ExprEvalResult eval_stmt_while(StmtWhile *stmt, Ctx &ctx) {
327 Interpreter::ExprEvalResult expr_result = Interpreter::eval_expr(stmt->m_expr.get(), ctx);
328 Interpreter::ExprEvalResult result{};
329
330 while (std::any_cast<bool>(expr_result.value()) == true) {
331 result = eval_stmt_block(stmt->m_block.get(), ctx);
332 if (result.value().has_value())
333 break;
334 expr_result = Interpreter::eval_expr(stmt->m_expr.get(), ctx);
335 }
336
337 return result;
338}

340Interpreter::ExprEvalResult eval_stmt_for(StmtFor *stmt, Ctx &ctx) {
341
342 Interpreter::ExprEvalResult result{};
343
344 Interpreter::ExprEvalResult start_expr = Interpreter::eval_expr(stmt->m_start.get(), ctx);
345 Interpreter::ExprEvalResult end_expr = Interpreter::eval_expr(stmt->m_end.get(), ctx);
346
347 EarlVar *enumerator = new EarlVar(stmt->m_enumerator.get(), EarlTy::Type::Int, false, std::any_cast<int>(start_expr.value()));
348 assert(!ctx.is_registered_earlvar(enumerator->m_id->lexeme()));
349 ctx.register_earlvar(enumerator);
350
351 while (std::any_cast<int>(enumerator->m_value) < std::any_cast<int>(end_expr.value())) {
352 result = eval_stmt_block(stmt->m_block.get(), ctx);
353
354 if (result.value().has_value())
355 break;
356
357 enumerator->m_value = std::any_cast<int>(enumerator->m_value)+1;
358 }
359
360 // Remove the enumerator as it is only a tmp variable.
361 ctx.deregister_earlvar(enumerator);
362 delete enumerator;
363
364 return result;
365}

367Interpreter::ExprEvalResult eval_stmt(Stmt *stmt, Ctx &ctx) {
368 switch (stmt->stmt_type()) {
369 case StmtType::Let: {
370 return eval_stmt_let(dynamic_cast<StmtLet *>(stmt), ctx);
371 } break;
372 case StmtType::Mut: {
373 return eval_stmt_mut(dynamic_cast<StmtMut *>(stmt), ctx);
374 } break;
375 case StmtType::Def: {
376 return eval_stmt_def(dynamic_cast<StmtDef *>(stmt), ctx);
377 } break;
378 case StmtType::Block: {
379 assert(false && "unimplemented");
380 } break;
381 case StmtType::Stmt_Expr: {
382 return eval_stmt_expr(dynamic_cast<StmtExpr *>(stmt), ctx);
383 } break;
384 case StmtType::If: {
385 return eval_stmt_if(dynamic_cast<StmtIf *>(stmt), ctx);
386 } break;
387 case StmtType::Stmt_Return: {
388 return eval_stmt_return(dynamic_cast<StmtReturn *>(stmt), ctx);
389 } break;
390 case StmtType::Stmt_While: {
391 return eval_stmt_while(dynamic_cast<StmtWhile *>(stmt), ctx);
392 } break;
393 case StmtType::Stmt_For: {
394 return eval_stmt_for(dynamic_cast<StmtFor *>(stmt), ctx);
395 } break;
396 default:
397 assert(false && "eval_stmt: invalid statement");
398 }
399 return Interpreter::ExprEvalResult{};
400}

402Interpreter::ExprEvalResult Interpreter::interpret(Program &program) {
403 Ctx ctx;
404
405 for (size_t i = 0; i < program.m_stmts.size(); ++i) {
406 eval_stmt(program.m_stmts.at(i).get(), ctx);
407 }
408
409 return Interpreter::ExprEvalResult{};
