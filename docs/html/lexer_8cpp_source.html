42void Lexer::append(Token *tok) {
43 if (!m_hd) {
44 m_hd = tok;
45 m_tl = tok;
46 } else {
47 m_tl->m_next = tok;

49 }

53Token *Lexer::peek(size_t n) {

56 tok = tok->m_next;

61Token *Lexer::next(void) {
62 if (!m_hd)
63 return nullptr;
64
65 Token *tok = m_hd;
66 m_hd = tok->m_next;
67
68 if (!m_hd)
69 m_tl = nullptr;
70
71 --m_len;

90static size_t consume_until(const std::string &s, const std::function<bool(char)> &predicate) {
91 size_t i;
92 bool skip = false;
93 for (i = 0; s[i]; ++i) {
94 if (!skip && predicate(s[i])) {
95 return i;
96 }
97 if (s[i] == '\\') {
98 skip = true;
99 }
100 else {
101 skip = false;
102 }
103 }
104 return i;
105}

117static char *read_file(const char *filepath) {
118 FILE *f = fopen(filepath, "rb");
119
120 if (f == NULL || fseek(f, 0, SEEK_END)) {
121 ERR_WARGS(ErrType::Fatal, "could not find the specified source filepath: %s", filepath);
122 }
123
124 long length = ftell(f);
125 rewind(f);
126 if (length == -1 || (unsigned long) length >= SIZE_MAX) {
127 return NULL;
128 }
129
130 size_t ulength = (size_t)length;
131 char *buffer = (char *)malloc(ulength+1);
132
133 if (buffer == NULL || fread(buffer, 1, ulength, f) != ulength) {
134 free(buffer);
135 return NULL;
136 }
137 buffer[ulength] = '\0';
138
139 return buffer;
140}

164static bool issym(char c) {
165 return !isalnum(c) && c != '_';

168static bool try_comment(char *src, std::string &comment) {
169 if (std::string(src).compare(0, comment.length(), comment) == 0)
170 return find_comment_end(src);
171 return false;

174Lexer lex_file(const char *filepath, std::vector<std::string> &keywords, std::vector<std::string> &types, std::string &comment) {
175 std::string src = read_file(filepath);
176
177 Lexer lexer;
178
179 const std::unordered_map<std::string, TokenType> ht = {
180 {"(", TokenType::Lparen},
181 {")", TokenType::Rparen},
182 {"[", TokenType::Lbracket},
183 {"]", TokenType::Rbracket},
184 {"{", TokenType::Lbrace},
185 {"}", TokenType::Rbrace},
186 {"#", TokenType::Hash},
187 {".", TokenType::Period},
188 {";", TokenType::Semicolon},
189 {",", TokenType::Comma},
190 {">", TokenType::Greaterthan},
191 {"<", TokenType::Lessthan},
192 {"=", TokenType::Equals},
193 {"&", TokenType::Ampersand},
194 {"*", TokenType::Asterisk},
195 {"+", TokenType::Plus},
196 {"-", TokenType::Minus},
197 {"/", TokenType::Forwardslash},
198 {"|", TokenType::Pipe},
199 {"^", TokenType::Caret},
200 {"?", TokenType::Questionmark},
201 {"\\", TokenType::Backwardslash},
202 {"!", TokenType::Bang},
203 {"@", TokenType::At},
204 {"$", TokenType::Dollarsign},
205 {"%", TokenType::Percent},
206 {"`", TokenType::Backtick},
207 {"~", TokenType::Tilde},
208 {":", TokenType::Colon},
209 {"&&", TokenType::Double_Ampersand},
210 {"||", TokenType::Double_Pipe},
211 {">=", TokenType::Greaterthan_Equals},
212 {"<=", TokenType::Lessthan_Equals},
213 {"==", TokenType::Double_Equals},
214 {"!=", TokenType::Bang_Equals},
215 {"+=", TokenType::Plus_Equals},
216 {"-=", TokenType::Minus_Equals},
217 {"*=", TokenType::Asterisk_Equals},
218 {"/=", TokenType::Forwardslash_Equals},
219 {"%=", TokenType::Percent_Equals},
220 {"->", TokenType::RightArrow},
221 {"..", TokenType::Double_Period},
222 };
223
224 size_t row = 0, col = 0, i = 0;
225 while (src[i]) {
226 char *lexeme = &src[i];
227 char c = src[i];
228
229 if (c == comment[0]) {
230 size_t comment_len;
231 if ((comment_len = try_comment(lexeme, comment)) >= comment.size()) {
232 while (src[i++] != '\n');
233 col = 1;
234 row += 1;
235 continue;
236 }
237 }
238 else {
239 lexeme = &src[i];
240 }
241
242 // Newlines
243 if (c == '\r' || c == '\n') {
244 ++row;
245 col = 1;
246 ++i;
247 }
248
249 // Tabs/empty spaces
250 else if (c == '\t' || c == ' ') {
251 ++col;
252 ++i;
253 }
254
255 // String literal
256 else if (c == '"') {
257 size_t strlit_len = consume_until(lexeme+1, [](const char c) {
258 return c == '"';
259 });
260 Token *tok = token_alloc(lexer, lexeme+1, strlit_len, TokenType::Strlit, row, col, filepath);
261 lexer.append(tok);
262 i += 1 + strlit_len + 1;
263 col += 1 + strlit_len + 1;
264 }
265
266 // Character literal
267 else if (c == '\'') {
268 lexeme += 1;
269 Token *tok = token_alloc(lexer, lexeme, 1, TokenType::Charlit, row, col, filepath);
270 lexer.append(tok);
271 i += 3;
272 col += 1;
273 }
274
275 // Identifers/Keywords
276 else if (isalpha(c) || c == '_') {
277 size_t ident_len = consume_until(lexeme, [](char c) {
278 return !(c == '_' || isalnum(c));
279 });
280
281 Token *tok = nullptr;
282
283 bool is_ty = is_type(lexeme, ident_len, types);
284 bool is_kw = is_keyword(lexeme, ident_len, keywords);
285
286 if (is_ty) {
287 tok = token_alloc(lexer, lexeme, ident_len, TokenType::Type, row, col, filepath);
288 }
289 else if (is_kw) {
290 tok = token_alloc(lexer, lexeme, ident_len, TokenType::Keyword, row, col, filepath);
291 }
292 else {
293 tok = token_alloc(lexer, lexeme, ident_len, TokenType::Ident, row, col, filepath);
294 }
295
296 lexer.append(tok);
297 i += ident_len;
298 col += ident_len;
299 }
300
301 // Numbers
302 else if (isdigit(c)) {
303 size_t intlit_len = consume_until(lexeme, [](char c) {
304 return !isdigit(c);
305 });
306 Token *tok = token_alloc(lexer, lexeme, intlit_len, TokenType::Intlit, row, col, filepath);
307 lexer.append(tok);
308 i += intlit_len;
309 col += intlit_len;
310 }
311
312 // Symbols
313 else {
314 std::string buf;
315 for (size_t j = 0; src[i+j] && issym(src[i+j]); ++j) {
316 if (src[i+j] != ' ' && src[i+j] != '\t' && src[i+j] != '\n' && src[i+j] != '\t') {
317 buf.push_back(src[i+j]);
318 }
319 }
320
321 while (!buf.empty()) {
322 auto it = ht.find(buf);
323 if (it != ht.end()) {
324 Token *tok = token_alloc(lexer, lexeme, buf.size(), (*it).second, row, col, filepath);
325 lexer.append(tok);
326 col += buf.size();
327 i += buf.size();
328 break;
329 }
330 buf.pop_back();
331 }
332 }
333 }
334
335 lexer.append(token_alloc(lexer, nullptr, 0, TokenType::Eof, row, col, filepath));
336 return lexer;
337}
